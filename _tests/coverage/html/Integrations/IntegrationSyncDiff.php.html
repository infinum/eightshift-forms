430                'message' => \esc_html__('Integration inner ID is different than content inner ID.', 'eightshift-forms'),
431            ];
432        }
433
434        // Run diff.
435        $output = $this->diffChanges($integration, $content, $editorOutput);
436
437        // In case of integration errors resulting in empty output, we will not delete the form. Instead, we will allow the user to decide what to do next.
438        if ($output['isOutputMissing']) {
439            return [
440                'formId' => $formId,
441                'status' => UtilsConfig::STATUS_ERROR,
442                'debugType' => 'after_empty_output',
443                'message' => \esc_html__('It appears that there is an error with the API connection when retrieving form data as the output is currently empty.', 'eightshift-forms'),
444                'data' => $output,
445            ];
446        }
447
448        // Bailout if update is not necesery.
449        if (!$output['update']) {
450            return [
451                'formId' => $formId,
452                'status' => UtilsConfig::STATUS_SUCCESS,
453                'debugType' => 'after_no_update',
454                'message' => \esc_html__('Integration and local form are the same, no update required.', 'eightshift-forms'),
455                'data' => $output,
456            ];
457        }
458
459        // Bailout if we have output to show.
460        if ($output['output']) {
461            // Bailout if db content update with success.
462            return [
463                'formId' => $formId,
464                'status' => UtilsConfig::STATUS_SUCCESS,
465                'debugType' => 'after_success',
466                'message' =>  \esc_html__('Form updated.', 'eightshift-forms'),
467                'data' => $output,
468            ];
469        }
470
471        // Bailout if some undefined error occurred.
472        return [
473            'formId' => $formId,
474            'status' => UtilsConfig::STATUS_ERROR,
475            'debugType' => 'after_undefined',
476            'message' => \esc_html__('Something went wrong.', 'eightshift-forms'),
477            'data' => $output,
478        ];
479    }

481    /**
482     * Compare content and integration form and outputs diff of a built block ready for import in the db.
483     *
484     * @param array<string, mixed> $integration Form integration content.
485     * @param array<string, mixed> $content Form content.
486     * @param boolean $editorOutput Change block output format depending on the usage.
487     *
488     * @return array<string, mixed>
489     */
490    private function diffChanges(array $integration, array $content, bool $editorOutput = false): array
491    {
492        // Prepare arrays for diff.
493        $diff = $this->prepareContentBlocksForCheck($content['fields']['innerBlocks'][0]['innerBlocks'] ?? [], $this->prepareIntegrationBlocksForCheck($integration['fields']));
494
495        $integrationType = $integration['type'] ?? '';
496        $diffOrder = $diff['order'] ?? [];
497        $diffOrderInner = $diff['orderInner'] ?? [];
498
499        // Prepare standard output.
500        $output = [
501            'type' => $integrationType,
502            'itemId' => $integration['itemId'] ?? '',
503            'innerId' => $integration['innerId'] ?? '',
504            'typeAttrs' => $content['fields']['innerBlocks'][0]['attrs'] ?? [],
505            'update' => false,
506            'removed' => [],
507            'added' => [],
508            'replaced' => [],
509            'changed' => [],
510            'order' => $diffOrder,
511            'orderInner' => $diffOrderInner,
512            'output' => [],
513            'isOutputMissing' => false,
514            'diff' => $diff['diff'],
515        ];
516
517        // Loop diff of content and integration.
518        foreach ($diff['diff'] as $key => $block) {
519            $isExternal = $block['content']['isExternal'] ?? false;
520
521            if ($isExternal) {
522                // Output none forms blocks.
523                $changes = [
524                    'removed' => [],
525                    'added' => [],
526                    'replaced' => [],
527                    'changed' => [],
528                    'output' => $block['content'],
529                    'update' => false,
530                ];
531            } else {
532                // Do diff on one field.
533                $changes = $this->diffChange($block['integration'] ?? [], $block['content'] ?? [], $key);
534            }
535
536            // Loop all outputs and prepare for the final output.
537            foreach ($changes as $changeKey => $changeValue) {
538                // No value no output necesery.
539                if (!$changeValue) {
540                    continue;
541                }
542
543                // Some output keys are array, some are strings and some require keys.
544                switch ($changeKey) {
545                    case 'update':
546                        $output[$changeKey] = $changeValue;
547                        break;
548                    case 'output':
549                        $output[$changeKey][$key] = $changeValue;
550                        break;
551                    default:
552                        $output[$changeKey][] = $changeValue;
553                        break;
554                }
555            }
556        }
557
558        // Loop top level block and output attributes but ignore IDs for integration.
559        if ($output['typeAttrs']) {
560            $typeAttrs = [];
561
562            foreach ($output['typeAttrs'] as $attrKey => $attrValue) {
563                if ($attrKey === "{$output['type']}IntegrationId" || $attrKey === "{$output['type']}IntegrationInnerId") {
564                    continue;
565                }
566
567                $typeAttrs[$attrKey] = $attrValue;
568            }
569
570            $output['typeAttrs'] = $typeAttrs;
571        }
572
573        // If output is missing create a flag and bailout.
574        if (!$output['output']) {
575            $output['isOutputMissing'] = true;
576        } else {
577            // If output not is missing recreate a form.
578            // Recounstruct blocks output and build array for final serialization.
579            $output['output'] = $this->reconstructBlocksTopLevelOutput($output, $editorOutput);
580        }
581
582        return $output;
583    }

585    /**
586     * Compare integration and content of one field and outputs new field.
587     *
588     * @param array<string, mixed> $integration Form integration content.
589     * @param array<string, mixed> $content Form content.
590     * @param string $key Index key.
591     *
592     * @return array<string, mixed>
593     */
594    private function diffChange(array $integration, array $content, string $key): array
595    {
596        // Prepare output.
597        $output = [
598            'removed' => [],
599            'added' => [],
600            'replaced' => [],
601            'changed' => [],
602            'output' => [],
603            'update' => false,
604        ];
605
606        // We need to add default attributes to the content data as attributes that have default values will not not ba added in this list and they will fall on sync.
607        // We don't need to remove default attributes at the end as Block Editor will do this for us.
608        if (isset($content['component'])) {
609            $content['attrs'] = \array_merge(
610                $this->getAttrsDefault($content['component']),
611                $content['attrs'],
612            );
613        }
614
615        // Remove item if block is not present on integration, output nothing.
616        if (!$integration) {
617            // Skip removing step block because it doesn't exist on the integration.
618            if (isset($content['component']) && $content['component'] === Helpers::getBlock('step')['blockName']) {
619                $output['update'] = false;
620                $output['output'] = $content;
621                return $output;
622            }
623
624            $output['update'] = true;
625            $output['removed'] = $key;
626            return $output;
627        }
628
629        // If field exists on the integration but not on the content add it.
630        if (!$content) {
631            $output['update'] = true;
632            $output['added'] = $key;
633            $output['output'] = $integration;
634            return $output;
635        }
636
637        // If field type has changed on integration use the integration one.
638        if ($integration['component'] !== $content['component']) {
639            $output['update'] = true;
640            $output['replaced'] = $key;
641            $output['output'] = $integration;
642            return $output;
643        }
644
645        // Check if disabled attrs changed.
646        $innerOutput = $content;
647
648        // Find prefix of the component.
649        $prefix = Helpers::kebabToCamelCase($integration['component'] . \ucfirst($integration['component']));
650
651        // Find components disabled options.
652        $disabledOptionsIntegration = $integration['attrs']["{$prefix}DisabledOptions"] ?? [];
653
654        $innerOutput['attrs']["{$prefix}DisabledOptions"] = $disabledOptionsIntegration;
655
656        // Check disabled options.
657        if ($disabledOptionsIntegration) {
658            foreach ($disabledOptionsIntegration as $disabledOption) {
659                // Find attributes in integration and content that match disabled options item.
660                $i = $integration['attrs'][$disabledOption] ?? '';
661                $c = $content['attrs'][$disabledOption] ?? '';
662                // If intregration is missing disabled or protected attribute. There could be and issue in the mapping of component attributes for integration.
663                if (!$i) {
664                    // This condition is here if both integration and content are empty.
665                    if ($i !== $c) {
666                        $output['update'] = true;
667                        $output['replaced'] = $key;
668                        $output['output'] = $integration;
669                        break;
670                    }
671                }
672
673                // If content has missing, disabled or protected attribute add it from integration.
674                if (!$c) {
675                    // This condition is here if both integration and content are empty.
676                    if ($c !== $i) {
677                        $output['update'] = true;
678                        $output['changed'][$key][] = $disabledOption;
679                        $innerOutput['attrs'][$disabledOption] = $i;
680                        break;
681                    }
682                }
683
684                // If values of attribute in content and intregation are diffrerent do something.
685                if ($i !== $c) {
686                    // If protected attribute name has changed we need to update the whole block. This is an unlikely scenario but it can happen.
687                    if ($i === "{$prefix}Name" && $c === "{$prefix}Name") {
688                        $output['update'] = true;
689                        $output['replaced'] = $key;
690                        $output['output'] = $integration;
691                        break;
692                    }
693
694                    // Output the changed value.
695                    $output['update'] = true;
696                    $output['changed'][$key][] = $disabledOption;
697                    $innerOutput['attrs'][$disabledOption] = $i;
698                    continue;
699                }
700            }
701        }
702
703        // Populate output data.
704        $output['output'] = $innerOutput;
705
706        $disabledOptionsKeys = \array_flip($output['output']['attrs']["{$prefix}DisabledOptions"]);
707
708        // Add missing content attributes from integration.
709        $missingAttributes = \array_diff_key($integration['attrs'], $content['attrs']);
710        if ($missingAttributes) {
711            foreach ($missingAttributes as $missingAttributesKey => $missingAttributesValue) {
712                // No need to add default values.
713                if ($missingAttributesKey === 'inputInputType' && $missingAttributesValue === 'text') {
714                    continue;
715                }
716
717                // Add missing attributes to the otput.
718                $output['update'] = true;
719                $output['added'] = $missingAttributesKey;
720                $output['output']['attrs'][$missingAttributesKey] = $missingAttributesValue;
721            }
722        }
723
724        // Remove attributes removed from integration but it is still in the content.
725        $removedAttributes = \array_diff_key($content['attrs'], $integration['attrs']);
726        if ($removedAttributes) {
727            foreach ($removedAttributes as $removedAttributesKey => $removedAttributesValue) {
728                if (!isset($disabledOptionsKeys[$removedAttributesKey])) {
729                    continue;
730                }
731
732                // Remove attributes to the otput.
733                $output['update'] = true;
734                $output['removed'] = $removedAttributesKey;
735                unset($output['output']['attrs'][$removedAttributesKey]);
736            }
737        }
738
739        return $output;
740    }

742    /**
743     * Prepare integration blocks for diff check.
744     *
745     * @param array<string, mixed> $blocks Blocks from external integration.
746     *
747     * @return array<string, mixed>
748     */
749    private function prepareIntegrationBlocksForCheck(array $blocks): array
750    {
751        $output = [];
752
753        $nestedKeys = \array_flip(AbstractFormBuilder::NESTED_KEYS);
754        $namespace = Helpers::getSettingsNamespace();
755
756        foreach ($blocks as $block) {

758

760                continue;
761            }
762
763            $blockType = Helpers::kebabToCamelCase($blockTypeOriginal, '-');
764            $blockName = "{$blockType}Name";
765
766            $name = $block[$blockName] ?? '';
767
768            if (!$name) {
769                continue;
770            }
771
772            $output[$name]['integration']  = [

774                'component' => $blockTypeOriginal,

778                'inner' => [],
779                'isExternal' => false,
780            ];

783
784            if ($innerBlocks) {

789                        continue;
790                    }

794                    $innerPrefix = $blockInnerType . \ucfirst($blockInnerType);
795
796                    $output[$this->getInnerBlocksKeyName($innerPrefix, $blockInnerAttributes, $innerKey, $name)]['integration']  = [
797                        'namespace' => $namespace,
798                        'component' => $blockInnerTypeOriginal,
799                        'prefix' => $innerPrefix,
800                        'attrs' => $blockInnerAttributes,
801                        'parent' => $name,
802                        'inner' => [],
803                        'isExternal' => false,
804                    ];
805                }
806            }
807        }
808
809        return $output;
810    }

812    /**
813     * Prepare content blocks for diff check and combine with integration blocks.
814     *
815     * @param array<string, mixed> $blocks Blocks from form content.

817     *
818     * @return array<string, mixed>
819     */
820    private function prepareContentBlocksForCheck(array $blocks, array $integration): array

824        // Used to preserver content order of the blocks.

826        $orderInner = [];

831            if (!$blockTypeOriginal) {

846                    'attrs' => $block['attrs'],
847                    'parent' => '',
848                    'inner' => $block['innerBlocks'],
849                    'isExternal' => true,
850                ];

852                $order[] = $noneFormsBlockName;
853
854                continue;
855            }
856
857            $blockName = $blockType['prefix'] . "Name";

861            }

904                        'isExternal' => false,
905                    ];

908                }

910        }

912        return [

916        ];

919    /**
920     * Rebuild for blocks output in block grammar format after diff check. Top level with all blocks.
921     *
922     * @param array<string, mixed> $data Diff prepared data.
923     * @param boolean $editorOutput Change output keys depending on the output type.
924     *
925     * @return array<int, array<string, array<int|string, array<string, mixed>|string>|string>>
926     */
927    private function reconstructBlocksTopLevelOutput(array $data, bool $editorOutput = false): array
928    {
929        $integrationType = $data['type'] ?? '';
930        $blockOutput = $data['output'] ?? [];
931
932        // Reconstruct fields output and provide a custom order.
933        $fieldsOutput = $this->reconstructBlocksOutput(
934            $this->orderBlocks(
935                $blockOutput,
936                [
937                    'order' => $data['order'] ?? [],
938                    'orderInner' => $data['orderInner'] ?? [],
939                ],
940                $integrationType
941            ),
942            $editorOutput
943        );
944
